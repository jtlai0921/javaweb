SELECT d.DISCOUNT_CODE FROM APP.DISCOUNT_CODE d 
WHERE d.RATE >= 10;

SELECT * FROM APP.CUSTOMER c
WHERE c.DISCOUNT_CODE in ('H', 'M');

SELECT * FROM APP.CUSTOMER c
WHERE c.DISCOUNT_CODE in (SELECT d.DISCOUNT_CODE FROM APP.DISCOUNT_CODE d WHERE d.RATE >= 10)

-- 總金額
SELECT SUM(pu.QUANTITY * pr.PURCHASE_COST) as subtotal 
FROM APP.PURCHASE_ORDER pu, APP.PRODUCT pr
WHERE pu.CUSTOMER_ID = 36 AND pu.PRODUCT_ID = pr.PRODUCT_ID;

-- 總金額 * 折扣
SELECT SUM(pu.QUANTITY * pr.PURCHASE_COST) * (100-(SELECT d.RATE FROM APP.DISCOUNT_CODE d WHERE d.DISCOUNT_CODE in (SELECT c.DISCOUNT_CODE FROM APP.CUSTOMER c WHERE c.CUSTOMER_ID = 36)))/100 as subtotal 
FROM APP.PURCHASE_ORDER pu, APP.PRODUCT pr
WHERE pu.CUSTOMER_ID = 36 AND pu.PRODUCT_ID = pr.PRODUCT_ID;

-- View : PU_VIEW
SELECT 
    pu.ORDER_NUM, cu."NAME" as CUSTOMER_NAME, pu.PRODUCT_ID, pr.DESCRIPTION as PRODUCT_NAME,
    pr.PURCHASE_COST, di.RATE, pu.QUANTITY,
    CAST((100-di.RATE)/100 * pr.PURCHASE_COST * pu.QUANTITY as DOUBLE) as SUBTOTAL
FROM 
    APP.PURCHASE_ORDER pu
LEFT JOIN APP.CUSTOMER cu ON pu.CUSTOMER_ID = cu.CUSTOMER_ID
LEFT JOIN APP.PRODUCT pr ON pu.PRODUCT_ID = pr.PRODUCT_ID
LEFT JOIN APP.DISCOUNT_CODE di ON cu.DISCOUNT_CODE = di.DISCOUNT_CODE
